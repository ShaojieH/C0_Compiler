int Fibonacci(int n) {
    if (n == 1) {
        return(1);
    }
    if (n == 2) {
        return(1);
    }
    return(Fibonacci(n - 1) + Fibonacci(n - 2));
}

void test_recursion(){
	if(Fibonacci(10) == 55){
		printf("recursion right");
	} else{
		printf("recursion error");
	}
}

void main(){
	int a;
	scanf(a);
	test_recursion();
	printf(a);
}
